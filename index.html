<script>

// localStorage-backed storage (works on GitHub Pages / any static host; per-browser)

window.storage = {

  get: async (k)=>{ const v=localStorage.getItem(k); return v!==null?{key:k,value:v}:null; },

  set: async (k,v)=>{ localStorage.setItem(k,v); return {key:k,value:v}; },

  delete: async (k)=>{ localStorage.removeItem(k); return {key:k,deleted:true}; },

  list: async ()=>({keys:Object.keys(localStorage)})

};


const KEY='sanjeev-road-to-marriage-v1';

const NAME_KEY='sanjeev-teammate-name';


const M=[

  {t:'Operation Nickname',     d:'Give your fiancée an adorable nickname and reveal it to the team.'},

  {t:'Words of Affirmation',   d:'Send her a sweet good-morning text before 9 AM.'},

  {t:'Master Chef Trainee',    d:'Find out her favourite dish and admit if you can cook it.'},

  {t:'Surprise Engineer',      d:'Plan one small surprise for the weekend.'},

  {t:'Compliment Combo',       d:'Give her 3 genuine compliments today.'},

  {t:'Picture Perfect',        d:'Take a cute selfie together and share it (with permission).'},

  {t:'DJ Husband',             d:'Learn the lyrics to her favourite song.'},

  {t:'Honeymoon HQ',           d:'Shortlist two honeymoon destinations.'},

  {t:'The Gratitude List',     d:'Write down 5 things you love about her.'},

  {t:'First Dance Rehearsal',  d:'Practise one first-dance move.'},

  {t:'Gift Logistics',         d:'Plan or buy a small surprise gift.'},

  {t:'Sorry in Advance',       d:'Master a sincere "I\'m sorry."'},

  {t:'Vows Draft',             d:'Write the first line of your vows.'},

  {t:'Grand Finale',           d:'Declare to the team you\'re 100% ready for married life!'}

];


let state={days:{}};

let myName='';


function ensureDays(){ for(let i=1;i<=14;i++){ if(!state.days[i]) state.days[i]={status:null,rating:null,by:null}; } }

function dayDate(i){ return new Date(2026,5,2+i); } // Day 1 = Jun 3 2026

function fmt(d){ return d.toLocaleDateString('en-US',{weekday:'short',month:'short',day:'numeric'}); }

function startOfDay(d){ return new Date(d.getFullYear(),d.getMonth(),d.getDate()); }

function isLocked(i){ return startOfDay(dayDate(i)) > startOfDay(new Date()); }


async function load(){

  try{ const r=await window.storage.get(KEY,true); if(r&&r.value) state=JSON.parse(r.value); }catch(e){}

  ensureDays();

  try{ const n=await window.storage.get(NAME_KEY,false); if(n&&n.value){ myName=n.value; } }catch(e){}

  const ni=document.getElementById('name'); ni.value=myName; updateWho();

  render();

  document.getElementById('loading').style.display='none';

}

async function save(){

  try{ await window.storage.set(KEY,JSON.stringify(state),true); }

  catch(e){ console.error('save failed',e); }

}

async function saveName(){ try{ await window.storage.set(NAME_KEY,myName,false); }catch(e){} }


function carriedInto(i){

  const out=[];

  for(let m=1;m<i;m++){ if(state.days[m].status==='skipped') out.push(m); }

  return out;

}

function needName(){

  if(myName.trim()) return true;

  const ni=document.getElementById('name'); ni.focus();

  ni.style.borderColor='#c0392b';

  setTimeout(()=>ni.style.borderColor='',900);

  return false;

}

async function act(i,kind){

  if(isLocked(i)) return;

  if(!needName()) return;

  const day=state.days[i];

  if(kind==='done'){ day.status='done'; if(day.rating===null) day.rating=8; day.by=myName; }

  else { day.status='skipped'; day.rating=2; day.by=myName; }

  await save(); render();

  if(i===14 && kind==='done') confetti();

}

async function setRating(i,n){

  const day=state.days[i];

  if(day.status!=='done') return;

  day.rating=n; await save(); render();

}

async function clearDay(i){

  if(isLocked(i)) return;

  state.days[i]={status:null,rating:null,by:null};

  await save(); render();

}

function stats(){

  let done=0,carry=0,sum=0,cnt=0;

  for(let i=1;i<=14;i++){ const d=state.days[i];

    if(d.status==='done'){done++; sum+=d.rating; cnt++;}

    else if(d.status==='skipped'){carry++; sum+=2; cnt++;}

  }

  const avg=cnt?sum/cnt:0;

  return {done,carry,rem:14-done-carry,avg,cnt};

}

function verdictFor(avg,cnt){

  if(!cnt) return ['Not started — fiancée is waiting 👀',0];

  const pct=Math.round(avg*10);

  if(avg<3) return ['🚨 Emergency Husband Bootcamp',pct];

  if(avg<5) return ['🛠️ Needs Serious Training',pct];

  if(avg<7) return ['🌱 Showing Some Promise',pct];

  if(avg<9) return ['💍 Husband Material Loading…',pct];

  return ['🏆 Certified Husband Material',pct];

}

function render(){

  const s=stats();

  document.getElementById('s-avg').textContent=s.cnt?s.avg.toFixed(1):'—';

  document.getElementById('s-done').textContent=s.done;

  document.getElementById('s-carry').textContent=s.carry;

  document.getElementById('s-rem').textContent=s.rem;

  const [v,pct]=verdictFor(s.avg,s.cnt);

  document.getElementById('verdict').textContent=v;

  document.getElementById('meter').style.width=pct+'%';

  const grid=document.getElementById('grid'); grid.innerHTML='';

  for(let i=1;i<=14;i++){

    const d=state.days[i], locked=isLocked(i), m=M[i-1];

    const card=document.createElement('div');

    card.className='card '+(locked?'locked':(d.status||''));

    let badge='';

    if(locked) badge='<span class="badge l">🔒 Locked</span>';

    else if(d.status==='done') badge='<span class="badge d">✓ Did it</span>';

    else if(d.status==='skipped') badge='<span class="badge s">⏭ Skipped</span>';

    const carried=carriedInto(i);

    let carryHtml='';

    if(carried.length){

      carryHtml='<div class="carry"><b>Carried forward:</b> '+carried.map(c=>M[c-1].t).join(' · ')+'</div>';

    }

    let body='';

    if(locked){

      body='<div class="lockmsg">🔒 Unlocks '+fmt(dayDate(i))+'</div>';

    } else if(d.status==='skipped'){

      body='<div class="rate"><div class="rl">Rating (locked)</div><div class="pips">'+pips(i,2,true)+'</div></div>'

         +'<div class="by">⏭ Skipped by '+esc(d.by)+' — auto 2/10</div>'

         +'<div class="actions"><button class="btn-skip" onclick="clearDay('+i+')">Undo</button></div>';

    } else if(d.status==='done'){

      body='<div class="rate"><div class="rl">Rate this mission</div><div class="pips">'+pips(i,d.rating,false)+'</div></div>'

         +'<div class="by">✓ Done by '+esc(d.by)+'</div>'

         +'<div class="actions"><button class="btn-skip" onclick="clearDay('+i+')">Undo</button></div>';

    } else {

      body='<div class="actions">'

         +'<button class="btn-did" onclick="act('+i+',\'done\')">Did it ✅</button>'

         +'<button class="btn-skip" onclick="act('+i+',\'skip\')">Skip ⏭</button></div>';

    }

    card.innerHTML=

      '<div class="top"><div><div class="daytag">DAY '+i+'</div><div class="date">'+fmt(dayDate(i))+'</div></div>'+badge+'</div>'

      +'<div class="title">'+esc(m.t)+'</div>'

      +'<div class="task">'+esc(m.d)+'</div>'

      +carryHtml+body;

    grid.appendChild(card);

  }

}

function pips(i,val,locked){

  let h='';

  for(let n=1;n<=10;n++){

    h+='<button class="pip'+(n<=val?' on':'')+(locked?' lock':'')+'" '

      +(locked?'':'onclick="setRating('+i+','+n+')"')+'>'+n+'</button>';

  }

  return h;

}

function esc(s){ return (s||'').replace(/[&<>"']/g,c=>({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c])); }

function updateWho(){

  const w=document.getElementById('who');

  w.textContent=myName.trim()?('Rating as '+myName):'';

}

document.getElementById('name').addEventListener('input',e=>{

  myName=e.target.value; updateWho(); saveName();

});

async function resetBoard(){

  if(!confirm('Reset the whole board? This clears all ratings.')) return;

  state={days:{}}; ensureDays(); await save(); render();

}

function confetti(){

  const c=document.getElementById('confetti'), x=c.getContext('2d');

  c.width=innerWidth; c.height=innerHeight;

  const cols=['#d4a017','#f4a300','#7a1f2b','#2e7d4f','#fff'];

  const P=[];

  for(let i=0;i<160;i++) P.push({x:Math.random()*c.width,y:-20-Math.random()*c.height,

    s:5+Math.random()*7,vy:2+Math.random()*4,vx:-2+Math.random()*4,

    rot:Math.random()*6,vr:-.2+Math.random()*.4,col:cols[i%cols.length]});

  let t=0;

  (function f(){

    x.clearRect(0,0,c.width,c.height); t++;

    P.forEach(p=>{ p.x+=p.vx; p.y+=p.vy; p.rot+=p.vr;

      x.save(); x.translate(p.x,p.y); x.rotate(p.rot); x.fillStyle=p.col;

      x.fillRect(-p.s/2,-p.s/2,p.s,p.s*1.6); x.restore(); });

    if(t<170) requestAnimationFrame(f); else x.clearRect(0,0,c.width,c.height);

  })();

}

load();
